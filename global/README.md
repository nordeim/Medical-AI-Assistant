## üöÄ Quick Start

### Initialize Global Framework

```javascript
const { GlobalOperationsFramework } = require('./global/operations/global-operations-framework');
const { InternationalLocalizationFramework } = require('./global/localization/international-localization-framework');
const { CrossBorderComplianceFramework } = require('./global/compliance/cross-border-compliance-framework');
const { GlobalSupplyChainFramework } = require('./global/supply-chain/global-supply-chain-framework');
const { InternationalSupportFramework } = require('./global/support/international-support-framework');
const { GlobalPerformanceFramework } = require('./global/performance/global-performance-framework');
const { InternationalExpansionFramework } = require('./global/expansion/international-expansion-framework');

// Initialize all frameworks
const globalFramework = new GlobalOperationsFramework();
const localizationFramework = new InternationalLocalizationFramework();
const complianceFramework = new CrossBorderComplianceFramework();
const supplyChainFramework = new GlobalSupplyChainFramework();
const supportFramework = new InternationalSupportFramework();
const performanceFramework = new GlobalPerformanceFramework();
const expansionFramework = new InternationalExpansionFramework();

// Initialize all frameworks
await Promise.all([
    globalFramework.initializeFramework(),
    localizationFramework.initializeFramework(),
    complianceFramework.initializeFramework(),
    supplyChainFramework.initializeFramework(),
    supportFramework.initializeFramework(),
    performanceFramework.initializeFramework(),
    expansionFramework.initializeFramework()
]);

console.log('‚úÖ All frameworks initialized successfully');
```


```javascript
// Ensure compliance for North American operations

});
```
